 package org.eclipse.rdf4j.rio.hdt;
 
 import java.io.IOException;
 
 /**
  * HDT DictionarySection part. Various encodings exist.
  *
  * @author Bart Hanssens
  */
 abstract class HDTDictionarySection extends HDTPart {
 	protected enum Type {
 		PLAIN(1),
 		FRONT(2),
 		HTFC(3),
 		FMINDEX(4),
 		REPAIRDAC(5),
 		HASHHUFF(6);
 
 		private final int value;
 
 		protected int getValue() {
 			return value;
 		}
 
 		Type(int value) {
 			this.value = value;
 		}
 	}
 
 	/**
 	 * Get the size
 	 *
 	 * @return
 	 */
 	protected abstract int size();
 
 	/**
 	 * Get the entry
 	 *
 	 * @param i zero-based index
 	 * @return
 	 */
 	protected abstract byte[] get(int i) throws IOException;
 
 	/**
 	 * Constructor
 	 *
 	 * @param pos  position
 	 * @param name name
 	 */
 	protected HDTDictionarySection(String name, long pos) {
 		super(name, pos);
 	}
 }
