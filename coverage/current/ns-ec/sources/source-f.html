 /**
  * A class with static methods to create SPARQL queries

 	/**

 	 *

 	 *
 	 * @return a new {@link SelectQuery}

 	 * @see <a href="http://www.w3.org/TR/2013/REC-sparql11-query-20130321/#select">SPARQL Select Query</a>

 	}
