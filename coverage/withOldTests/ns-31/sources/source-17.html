 public class StatementSource extends AbstractQueryModelNode {

 		LOCAL,

 		REMOTE_POSSIBLY
 	}
 
 	protected String id;
 	protected StatementSourceType type;

 		super();
 		this.id = name;
 		this.type = type;
 	}

 	}

 		sb.append(" (id=").append(id);
 
 		sb.append(", type=").append(type);
 
 		sb.append(")");
 
 		return sb.toString();
 
 	}

 		final int prime = 31;

 	}

 	}
